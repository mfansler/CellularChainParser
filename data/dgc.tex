% Description : A simple differential graded coalgebra.

C_{0}(X) = \{ v \}
C_{1}(X) = \{ a, b \}
C_{2}(X) = \{ aa, ab \}

\partial aa = b

\Delta v = v \otimes v
\Delta a = v \otimes a + a \otimes v
\Delta b = v \otimes b + b \otimes v
\Delta aa = v \otimes aa + a \otimes a + aa \otimes v
\Delta ab = v \otimes ab + a \otimes b + ab \otimes v
